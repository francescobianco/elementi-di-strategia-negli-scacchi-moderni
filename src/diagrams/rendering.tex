% Diagram rendering helpers for the new edition.
% The source of truth for positions is src/diagrams/diagrams.csv.

\newcommand{\bookdiagramfen}[2]{%
  \par\begin{center}
    \chessboard[
      setfen=#1,
      boardfontsize=16pt,
      labelfontsize=6pt,
      showmover=false,
      label=false,
      margin=false,
      linewidth=0.35pt,
      color=gray,
      colorwhitebackfields=white,
      colorblackbackfields=gray!18,
      pgfstyle=straightmove,
    ]%

    {\small\emph{diagramma #2}}%
  \end{center}\par%
}

\newcommand{\bookdiagramplaceholder}[3][1.42in]{%
  \par\begin{center}
    \fbox{\includegraphics[width=#1,height=#1]{#2}}%

    {\small\emph{diagramma #3}}%
  \end{center}\par%
}
